#version 330


layout( triangles ) in;
layout( triangle_strip, max_vertices = 240 ) out;
in vec2 texCoords[];


out vec2 the_texCoords[3];

void main(){
	int i;
	for( i = 0; i < gl_in.length(); ){
		the_texCoords[i] = texCoords[i];
		gl_Position = gl_in[0].gl_Position;
		EmitVertex();
	}
	EndPrimitive();
}
